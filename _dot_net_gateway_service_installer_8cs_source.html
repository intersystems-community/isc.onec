92  private static void ChangeStartParameters(string serviceName, string[] args) {
93  var key = Registry.LocalMachine
94  .OpenSubKey("System")
95  .OpenSubKey("CurrentControlSet")
96  .OpenSubKey("Services")
97  .OpenSubKey(serviceName, true);
98  const string SubKey = "ImagePath";
99  var imagePath = (string) key.GetValue(SubKey);
100 
101  /*-
102  * Group #1 is the path to EXE.
103  * Group #3 (if any) is the port number.
104  * Group #5 (if any) is the keepalive flag.
105  */
106  const string ImagePathPattern = "^\\\"?([^\\\"]+\\.[Ee][Xx][Ee])\\\"?(\\s+(\\d+)(\\s+([Tt][Rr][Uu][Ee]|[Ff][Aa][Ll][Ss][Ee]))?)?\\s*$";
107  if (!Regex.IsMatch(imagePath, ImagePathPattern)) {
108  /*
109  * Never.
110  */
111  WriteLine("ImagePath doesn't match the pattern.", ConsoleColor.Red);
112  return;
113  }
114 
115  var match = Regex.Match(imagePath, ImagePathPattern);
116  if (match.Groups.Count < 2) {
117  /*
118  * Never.
119  */
120  WriteLine("Matcher is missing the group #1.", ConsoleColor.Red);
121  return;
122  }
123 
124  key.SetValue(SubKey, '"' + match.Groups[1].Value + "\" " + string.Join(" ", args));
125 
126  Logger.Warn("\\System\\CurrentControlSet\\Services\\" + serviceName + "\\ImagePath is changed.\nOld value:" + imagePath + "\nNew value:" + key.GetValue(SubKey));
127  }
