583  // When setting the buffer, you need 3 parameters the first time you set it,
584  // which we did in the Init method. The first of the three parameters
585  // tells what byte array to use as the buffer. After we tell what byte array
586  // to use we do not need to use the overload with 3 parameters any more.
587  // (That is the whole reason for using the buffer block. You keep the same
588  // byte array as buffer always, and keep it all in one block.)

592

594  // the buffer or not. If it is larger than the buffer, then we will have
595  // to post more than one send operation. If it is less than or equal to the
596  // size of the send buffer, then we can accomplish it in one send op.

602  } else {

613  }

623  // ____________________________________________________________________________
624  // This method is called by I/O Completed() when an asynchronous send completes.
625  // If all of the data has been sent, then this method calls StartReceive
626  // to start another receive op on the socket to read any additional
627  // data sent from the client. If all of the data has NOT been sent, then it
628  // calls StartSend to send more data.
629  private void ProcessSend(SocketAsyncEventArgs receiveSendEventArgs) {
630  DataHoldingUserToken receiveSendToken = (DataHoldingUserToken) receiveSendEventArgs.UserToken;
631 
632  if (receiveSendEventArgs.SocketError == SocketError.Success) {
633  receiveSendToken.SendBytesRemainingCount = receiveSendToken.SendBytesRemainingCount - receiveSendEventArgs.BytesTransferred;
634 
635  if (receiveSendToken.SendBytesRemainingCount == 0) {
636  // If we are within this if-statement, then all the bytes in
637  // the message have been sent.
638  this.StartReceive(receiveSendEventArgs);
639  } else {
640  // If some of the bytes in the message have NOT been sent,
641  // then we will need to post another send operation, after we store
642  // a count of how many bytes that we sent in this send op.
643  receiveSendToken.BytesSentAlreadyCount += receiveSendEventArgs.BytesTransferred;
644 
645  // So let's loop back to StartSend().
646  this.StartSend(receiveSendEventArgs);
647  }
648  } else {
649  // If we are in this else-statement, there was a socket error.
650 
651  // We'll just close the socket if there was a
652  // socket error when receiving data from the client.
653  receiveSendToken.Reset();
654  this.CloseClientSocket(receiveSendEventArgs);
655  }
656  }

679  // If it has, then dataMessageReceived should be null.
680  if (receiveSendToken.DataHolder.DataMessageReceived != null) {
681  receiveSendToken.CreateNewDataHolder();
682  }
683

685  // to be used by another client. This

692  #pragma warning restore 420
